        if enemy:
            distance = hero.distanceTo(enemy)
            if distance < 5:
                if hero.isReady("cleave"):
                   hero.cleave(enemy)
                else:
                    hero.attack(enemy)
            else:
                hero.attack(enemy)
        else:
            break
