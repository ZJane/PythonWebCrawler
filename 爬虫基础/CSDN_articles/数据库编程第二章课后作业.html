INNER JOIN book b ON b.bid = p.rid

SELECT b.rid,b.willData,b.returnData,r.rName,k.bName FROM boorrow b 
INNER JOIN reader r ON r.rid=b.rid
INNER JOIN book k ON k.bid=b.nif
WHERE b.rid IN(SELECT rid FROM reader r WHERE r.rAddress IS NULL) 
ORDER BY b.willData DESC,b.rid DESC
