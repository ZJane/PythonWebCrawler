BottomNavigationBar底部导航实现

2018年08月07日 21:09:16
阅读数：4

首先导入相关依赖

implementation 'com.ashokvarma.android:bottom-navigation-bar:2.0.4'

 

fragment 页面（这里只是示例你想要几个页面就建几个fragment 但是别忘了建对应xml）

public class FragmentOne extends Fragment {
    @Nullable
    @Override
    public View onCreateView(@NonNull LayoutInflater inflater, @Nullable ViewGroup container, @Nullable Bundle savedInstanceState) {
        //R.layout.fragmentone显示对应的页面
        View view = inflater.inflate(R.layout.fragmentone,container,false);
        return view;
    }
}

接下来就是主页面了

public class HomeActivity extends AppCompatActivity implements BottomNavigationBar.OnTabSelectedListener {
    //BottomNavigationBar 
    private BottomNavigationBar mBottomNavigationBar;
    //fragment（这里我只用了3个页面）
    private FragmentOne mFragmentOne = new FragmentOne();
    private FragmentTwo mFragmentTwo = new FragmentTwo();
    private FragmentThree mFragmentThree = new FragmentThree();
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_home);
        navigation();
        //设置默认页面
        setDefaultFragment();
    }

    private void navigation() {
        mBottomNavigationBar = findViewById(R.id.bottom_navigation_bar);
        mBottomNavigationBar.setBackgroundStyle(BottomNavigationBar.BACKGROUND_STYLE_STATIC);
        mBottomNavigationBar.setMode(BottomNavigationBar.MODE_FIXED);

        mBottomNavigationBar.addItem(new BottomNavigationItem(R.drawable.a1, getString(R.string.home)).setInactiveIconResource(R.drawable.a2).setActiveColorResource(R.color.colortrue).setInActiveColorResource(R.color.colorfalse))
                .addItem(new BottomNavigationItem(R.drawable.c1, getString(R.string.order)).setInactiveIconResource(R.drawable.c2).setActiveColorResource(R.color.colortrue).setInActiveColorResource(R.color.colorfalse))
                .addItem(new BottomNavigationItem(R.drawable.d2, getString(R.string.my)).setInactiveIconResource(R.drawable.d1).setActiveColorResource(R.color.colortrue).setInActiveColorResource(R.color.colorfalse))
                .setFirstSelectedPosition(0)
                .initialise();
        mBottomNavigationBar.setTabSelectedListener(this);


    }

    @Override
    public void onTabSelected(int position) {
        FragmentTransaction transaction = getSupportFragmentManager().beginTransaction();


        switch (position) {
            case 0:

                transaction.replace(R.id.ll_content, mFragmentOne);
                break;

            case 1:

                transaction.replace(R.id.ll_content, mFragmentTwo);
                break;

            case 2:

                transaction.replace(R.id.ll_content, mFragmentThree);
                break;

            default:
                transaction.replace(R.id.ll_content, mFragmentOne);
                break;
        }
        transaction.commit();

    }

    @Override
    public void onTabUnselected(int position) {

    }

    @Override
    public void onTabReselected(int position) {

    }

    private void setDefaultFragment() {
        FragmentTransaction transaction = getSupportFragmentManager().beginTransaction();
        transaction.replace(R.id.ll_content, mFragmentOne);
        transaction.commit();
    }
}

<?xml version="1.0" encoding="utf-8"?>
<RelativeLayout

    xmlns:tools="http://schemas.android.com/tools"
    android:id="@+id/activity_main"
    android:layout_width="match_parent"
    android:layout_height="match_parent"
    tools:context=".activity.HomeActivity">

    <LinearLayout
        android:id="@+id/ll_content"
        android:layout_width="match_parent"
        android:layout_height="match_parent"
        android:orientation="vertical"/>

    <com.ashokvarma.bottomnavigation.BottomNavigationBar
        android:id="@+id/bottom_navigation_bar"
        android:layout_width="match_parent"
        android:layout_height="wrap_content"
        android:layout_alignParentBottom="true"/>
</RelativeLayout>
